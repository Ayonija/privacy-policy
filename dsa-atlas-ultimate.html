// ── LRU CACHE O(1) get/put ────────────────────────
class LRUCache {
    record Node(int key,int val){Node prev,next;}
    final int cap; final Map<Integer,Node> map=new HashMap<>();
    final Node head=new Node(0,0),tail=new Node(0,0);
    { head.next=tail; tail.prev=head; }

    int get(int k){
        if(!map.containsKey(k)) return -1;
        remove(map.get(k)); insertFront(map.get(k));
        return map.get(k).val;
    }
    void put(int k,int v){
        if(map.containsKey(k)) remove(map.get(k));
        Node n=new Node(k,v); map.put(k,n); insertFront(n);
        if(map.size()>cap){Node lru=tail.prev;remove(lru);map.remove(lru.key);}
    }
    void remove(Node n){n.prev.next=n.next;n.next.prev=n.prev;}
    void insertFront(Node n){n.next=head.next;n.prev=head;head.next.prev=n;head.next=n;}
}
Dummy head/tail trick: By having sentinel nodes at both ends, every insert/remove is interior — no null pointer special cases anywhere.

public String shortestPalindrome(String s){
    String rev=new StringBuilder(s).reverse().toString();
    String t=s+"#"+rev;  // # prevents overlap
    int[] kmp=new int[t.length()];
    for(int i=1;i<t.length();i++){
        int j=kmp[i-1];
        while(j>0&&t.charAt(i)!=t.charAt(j)) j=kmp[j-1];
        if(t.charAt(i)==t.charAt(j)) j++;

}

class LFUCache {
    int cap,minFreq=0;
    Map<Integer,Integer> val=new HashMap<>(),freq=new HashMap<>();
    Map<Integer,LinkedHashSet<Integer>> fKeys=new HashMap<>();

    int get(int k){
        if(!val.containsKey(k)) return -1;
        bump(k); return val.get(k);
    }
    void put(int k,int v){
        if(cap<=0) return;
        if(val.containsKey(k)){val.put(k,v);bump(k);return;}
        if(val.size()==cap){
            LinkedHashSet<Integer> lfu=fKeys.get(minFreq);
            int evict=lfu.iterator().next();
            lfu.remove(evict);val.remove(evict);freq.remove(evict);
        }
        val.put(k,v);freq.put(k,1);minFreq=1;
        fKeys.computeIfAbsent(1,x->new LinkedHashSet<>()).add(k);
    }
    void bump(int k){
        int f=freq.get(k); fKeys.get(f).remove(k);
        if(f==minFreq&&fKeys.get(f).isEmpty()) minFreq++;
        freq.put(k,f+1);
        fKeys.computeIfAbsent(f+1,x->new LinkedHashSet<>()).add(k);
    }
}
